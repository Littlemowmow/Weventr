## SECTION 3: APP ARCHITECTURE & NAVIGATION

### Entry Point
```swift
@main
struct SideQuestApp: App {
    @StateObject private var authViewModel = AuthViewModel()
    @StateObject private var appState = AppState()

    var body: some Scene {
        WindowGroup {
            RootView()
                .environmentObject(authViewModel)
                .environmentObject(appState)
                .preferredColorScheme(.dark)
        }
    }
}
```

### AppState
```swift
@MainActor
class AppState: ObservableObject {
    @Published var selectedTab: Tab = .discover
    @Published var hasCompletedOnboarding: Bool
    @Published var showJoinTripSheet: Bool = false
    @Published var navigateToTripId: UUID?
    @Published var activeTripSparkId: UUID?

    enum Tab: Int, CaseIterable {
        case discover = 0
        case trip = 1
        case profile = 2

        var title: String {
            switch self {
            case .discover: return "Discover"
            case .trip: return "Trips"
            case .profile: return "Profile"
            }
        }

        var icon: String {
            switch self {
            case .discover: return "safari.fill"
            case .trip: return "calendar.circle.fill"
            case .profile: return "person.circle.fill"
            }
        }
    }

    init() {
        self.hasCompletedOnboarding = UserDefaults.standard.bool(forKey: "hasCompletedOnboarding")
    }
}
```

### Navigation Flow
```
App Launch
    │
    ▼
RootView
    ├── isCheckingSession? → SplashView (logo + spinner)
    ├── !isAuthenticated → AuthView (login/signup)
    ├── !hasCompletedOnboarding → OnboardingContainerView (8 steps)
    └── Ready → MainTabView
                  ├── Tab 0: Discover (DiscoveryView)
                  ├── Tab 1: Trips (TripsView)
                  └── Tab 2: Profile (ProfileView)
```

```sql
-- ============================================
-- TRIP SPARKS (Phase 0 — Pre-Commitment)
-- ============================================
CREATE TABLE IF NOT EXISTS trip_sparks (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    creator_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
    destination TEXT NOT NULL,
    rough_date_start DATE,
    rough_date_end DATE,
    deadline TIMESTAMPTZ NOT NULL,
    shareable_code TEXT UNIQUE NOT NULL,  -- 8-char alphanumeric for link
    status TEXT NOT NULL DEFAULT 'collecting'
        CHECK (status IN ('collecting', 'expired', 'advanced', 'cancelled')),
    trip_id UUID REFERENCES trips(id),   -- set when advanced to planning
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS trip_spark_responses (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    trip_spark_id UUID NOT NULL REFERENCES trip_sparks(id) ON DELETE CASCADE,
    user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
    available_dates DATE[] NOT NULL,
    budget_accommodation NUMERIC(10,2),  -- per night max
    budget_transport NUMERIC(10,2),      -- round trip max
    budget_food NUMERIC(10,2),           -- per day max
    budget_activities NUMERIC(10,2),     -- per day max
    budget_total NUMERIC(10,2),          -- hard ceiling
    commitment TEXT NOT NULL CHECK (commitment IN ('in', 'maybe')),
    responded_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE(trip_spark_id, user_id)
);

-- RLS for trip_sparks
ALTER TABLE trip_sparks ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can view sparks they created or responded to" ON trip_sparks
    FOR SELECT USING (
        creator_id = auth.uid()
        OR id IN (SELECT trip_spark_id FROM trip_spark_responses WHERE user_id = auth.uid())
    );
CREATE POLICY "Users can create sparks" ON trip_sparks
    FOR INSERT WITH CHECK (creator_id = auth.uid());
CREATE POLICY "Creators can update their sparks" ON trip_sparks
    FOR UPDATE USING (creator_id = auth.uid());

-- RLS for trip_spark_responses
ALTER TABLE trip_spark_responses ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can view responses for their sparks" ON trip_spark_responses
    FOR SELECT USING (
        user_id = auth.uid()
        OR trip_spark_id IN (SELECT id FROM trip_sparks WHERE creator_id = auth.uid())
    );
CREATE POLICY "Users can submit responses" ON trip_spark_responses
    FOR INSERT WITH CHECK (user_id = auth.uid());
CREATE POLICY "Users can update their responses" ON trip_spark_responses
    FOR UPDATE USING (user_id = auth.uid());

-- ============================================
-- SWIPE ROUNDS (Phase 1 — Blind Match)
-- ============================================
CREATE TABLE IF NOT EXISTS swipe_rounds (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    trip_id UUID NOT NULL REFERENCES trips(id) ON DELETE CASCADE,
    created_by UUID NOT NULL REFERENCES auth.users(id),
    deadline TIMESTAMPTZ,
    default_to_yes BOOLEAN DEFAULT false,
    status TEXT NOT NULL DEFAULT 'active'
        CHECK (status IN ('active', 'completed', 'expired')),
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS activity_swipes (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    round_id UUID NOT NULL REFERENCES swipe_rounds(id) ON DELETE CASCADE,
    user_id UUID NOT NULL REFERENCES auth.users(id),
    activity_id UUID NOT NULL REFERENCES activities(id),
    direction TEXT NOT NULL CHECK (direction IN ('right', 'left', 'super')),
    swiped_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE(round_id, user_id, activity_id)
);

CREATE TABLE IF NOT EXISTS match_results (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    round_id UUID NOT NULL REFERENCES swipe_rounds(id) ON DELETE CASCADE,
    activity_id UUID NOT NULL REFERENCES activities(id),
    match_strength TEXT NOT NULL
        CHECK (match_strength IN ('squad_lock', 'majority', 'split', 'noped')),
    yes_count INTEGER NOT NULL DEFAULT 0,
    total_voters INTEGER NOT NULL DEFAULT 0,
    super_like_by UUID[],
    is_booked BOOLEAN DEFAULT false,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- RLS for swipe tables
ALTER TABLE swipe_rounds ENABLE ROW LEVEL SECURITY;
ALTER TABLE activity_swipes ENABLE ROW LEVEL SECURITY;
ALTER TABLE match_results ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Trip members can view rounds" ON swipe_rounds
    FOR SELECT USING (
        trip_id IN (SELECT trip_id FROM trip_members WHERE user_id = auth.uid())
        OR created_by = auth.uid()
    );
CREATE POLICY "Trip members can create rounds" ON swipe_rounds
    FOR INSERT WITH CHECK (
        trip_id IN (SELECT trip_id FROM trip_members WHERE user_id = auth.uid())
    );

CREATE POLICY "Users can view own swipes" ON activity_swipes
    FOR SELECT USING (user_id = auth.uid());
CREATE POLICY "Users can swipe" ON activity_swipes
    FOR INSERT WITH CHECK (user_id = auth.uid());

CREATE POLICY "Trip members can view results" ON match_results
    FOR SELECT USING (
        round_id IN (
            SELECT id FROM swipe_rounds WHERE trip_id IN (
                SELECT trip_id FROM trip_members WHERE user_id = auth.uid()
            )
        )
    );

-- ============================================
-- BUDGET LOCK (Phase 2)
-- ============================================
CREATE TABLE IF NOT EXISTS budget_locks (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    trip_id UUID NOT NULL REFERENCES trips(id) ON DELETE CASCADE,
    user_id UUID NOT NULL REFERENCES auth.users(id),
    accommodation_per_night NUMERIC(10,2),
    transport_round_trip NUMERIC(10,2),
    food_per_day NUMERIC(10,2),
    activities_per_day NUMERIC(10,2),
    total_cap NUMERIC(10,2),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE(trip_id, user_id)
);

CREATE TABLE IF NOT EXISTS expenses (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    trip_id UUID NOT NULL REFERENCES trips(id) ON DELETE CASCADE,
    paid_by UUID NOT NULL REFERENCES auth.users(id),
    amount NUMERIC(10,2) NOT NULL,
    currency TEXT DEFAULT 'USD',
    description TEXT NOT NULL,
    category TEXT CHECK (category IN ('accommodation', 'transport', 'food', 'activities', 'other')),
    split_type TEXT DEFAULT 'even' CHECK (split_type IN ('even', 'custom')),
    split_details JSONB,  -- for custom splits: {"user_id": amount}
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

ALTER TABLE budget_locks ENABLE ROW LEVEL SECURITY;
ALTER TABLE expenses ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Trip members can view budgets" ON budget_locks

CREATE POLICY "Users set own budget" ON budget_locks

CREATE POLICY "Users update own budget" ON budget_locks

CREATE POLICY "Trip members can view expenses" ON expenses
    FOR SELECT USING (

CREATE POLICY "Trip members can add expenses" ON expenses
```

## SECTION 6: THE FUNNEL (Non-Negotiable Flow)

```
Phase 0: Trip Spark → group commits dates + budget
Phase 1: Blind Match → group agrees on activities
Phase 2: Budget Lock → costs transparent + aligned
Phase 3: Squad Itinerary → plan built + bookings made
Phase 4: Active Trip → during travel (v2)
Phase 5: Reflection + Remix → content flywheel (v2)
```

**Rule: Every screen must belong to a phase. If it doesn't, ask which phase it serves.**

### AUTHENTICATION

**High Level:** Email + Apple Sign-In. Session persists in Keychain. New users auto-get a profile row.

**Screens:**

**AuthView** — Two modes: Login and Sign Up
- Login: email field, password field, "Sign In" button, "Forgot Password?" link, divider, "Sign in with Apple" button, "Don't have an account? Sign Up" toggle
- Sign Up: email, password, confirm password, "Create Account" button, same Apple/toggle
- Error display: inline red text below form
- Loading: button shows spinner, inputs disabled

**AuthViewModel:**
```swift
@MainActor
class AuthViewModel: ObservableObject {
    @Published var isAuthenticated = false
    @Published var isCheckingSession = true
    @Published var isLoading = false
    @Published var error: String?
    @Published var currentUser: User?
    @Published var currentProfile: Profile?
```

## SECTION 8: STATES EVERY SCREEN MUST HANDLE

Every screen needs these 4 states implemented:

| State | Pattern |
|-------|---------|
| **Loading** | `SQLoadingView(message: "...")` — centered spinner + message on sqBlack bg |
| **Empty** | `SQEmptyState(icon:title:message:actionTitle:action:)` — icon, title, body, optional CTA |
| **Error** | Red banner at top with error message + "Retry" button |
| **Content** | Normal view with data |

```swift
// Standard pattern for every view:
var body: some View {
    Group {
        if viewModel.isLoading {
            SQLoadingView(message: "Loading...")
        } else if let error = viewModel.error {
            ErrorBanner(message: error) { Task { await viewModel.retry() } }
        } else if viewModel.items.isEmpty {
            SQEmptyState(icon: "sparkles", title: "Nothing here yet", message: "...", actionTitle: "Get started") { }
        } else {
            contentView
        }
    }
    .task { await viewModel.load() }
}
```
